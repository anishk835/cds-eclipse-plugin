  ✅ WORKS - This will parse and validate correctly:

    namespace my.bookshop;

    type Currency = String(3);

    type Status : String enum { Active; Inactive; }

    entity Books {
      key ID: UUID not null;
      title: String(100) not null;
      price: Decimal(9,2) check price > 0;
      currency: Currency default 'USD';
      status: Status default #Active;
      stock: Integer;

      virtual isAvailable: Boolean = stock > 0;

      author: Association to Authors;
    }

    entity Authors {
      key ID: UUID;
      name: String(100) not null unique;
    }

    service CatalogService {
      entity Books as projection on Books {
        ID, title, price
      }
    }


  ❌ DOESN'T WORK - These critical features are missing:

    entity Orders {
      customer_ID: UUID;

      // ❌ ON condition not supported
      customer: Association to Customers
        on customer.ID = $self.customer_ID;

      // ❌ Action not supported
      action cancel(reason: String) returns Boolean;

      // ❌ Localized not supported
      description: localized String(500);
    }

    // ❌ SELECT query not supported
    entity Stats as SELECT from Orders {
      customer.ID,
      count(*) as orderCount
    } group by customer.ID;

    // ❌ Logical operators not supported
    entity ValidOrders {
      isValid: Boolean = status = 'active' and total > 0;
    }
